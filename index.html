<!DOCTYPE html>

<html>

<head>

  <title>HTML Files Index</title>

  <meta charset="utf-8">

  <style>

    body { font-family: -apple-system, BlinkMacSystemFont, "Segoe UI", Helvetica, Arial, sans-serif; margin: 2rem; }

    h1 { color: #24292e; }

    ul { list-style-type: none; padding: 0; }

    li { margin: 0.5rem 0; }

    a { text-decoration: none; color: #0366d6; }

    a:hover { text-decoration: underline; }

  </style>

</head>

<body>

  <h1>Available HTML Files</h1>

  <ul id="file-list">Loading...</ul>


  <script>

    fetch('.')

      .then(response => response.text())

      .then(data => {

        const parser = new DOMParser();

        const doc = parser.parseFromString(data, 'text/html');

        const links = doc.querySelectorAll('a');

        const ul = document.getElementById('file-list');

        ul.innerHTML = '';


        links.forEach(link => {

          const href = link.getAttribute('href');

          if (href && href.endsWith('.html') && href !== 'index.html') {

            const li = document.createElement('li');

            li.innerHTML = `<a href="${href}">${href}</a>`;

            ul.appendChild(li);

          }

        });


        if (ul.children.length === 0) {

          ul.innerHTML = '<li>No .html files found.</li>';

        }

      })

      .catch(() => {

        document.getElementById('file-list').innerHTML =

          'Unable to list files (directory listing may be disabled).';

      });

  </script>

</body>

</html>
